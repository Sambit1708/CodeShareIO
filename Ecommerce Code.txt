public boolean existsRecordForSameMonthAndYear(LocalDate inputDate) {
        List<FundDividendAmount> fundDividendAmountList = fundDividendAmountRepository.findAll();
        
        int inputMonth = inputDate.getMonthValue();
        int inputYear = inputDate.getYear();
        
        return fundDividendAmountList.stream()
            .anyMatch(record -> {
                LocalDate recordDate = record.getProjectedRecordDate();
                return recordDate != null && 
                       recordDate.getMonthValue() == inputMonth && 
                       recordDate.getYear() == inputYear;
            });
    }
